# [206-面试题24. 反转链表](https://leetcode-cn.com/problems/fan-zhuan-lian-biao-lcof/)

```js
var reverseList = function(head) {
	// 初始化前驱结点为 null, pre是反转链表的尾部，一步一步往头部添加
  let pre = null;
  // 初始化目标结点为头结点
  let cur = head;
  // 只要目标结点不为 null，遍历就得继续
  while (cur !== null) {
    // 记录一下 next 结点
    let next = cur.next;
    // 反转指针
    cur.next = pre;
    // pre 往前走一步
    pre = cur;
    // cur往前走一步
    cur = next;
  }
  // 反转结束后，pre 就会变成新链表的头结点
  return pre
};
```

这个题的理解，在链表继续的过程中，做处理。

链表的遍历：

```js
while (list) {
  list = list.next;
}
```

想要在链表的遍历做处理，就需要把这个过程拆开

```js
while (list) {
  temp = list.next;
  // handle listNode
  list = temp;
}
```

使用ES6的解构赋值

```js
var reverseList = function(head) {
    let [prev, current] = [null, head]
    while(current) {
        [current.next, prev, current] = [prev, current, current.next]
    }
    return prev
}
```

转义ES5

```js
"use strict";

var reverseList = function reverseList(head) {
  var prev = null,
      current = head;

  while (current) {
    var _ref = [prev, current, current.next];
    current.next = _ref[0];
    prev = _ref[1];
    current = _ref[2];
  }

  return prev;
};
```

```js
var reverseList = function(head) {
  // 递归结束条件
  if (head === null || head.next === null) {
    return head
  }

  // 递归反转 子链表
  let newReverseList = reverseList(head.next)
  // 获取原来链表的第2个节点newReverseListTail
  let newReverseListTail = head.next
  // 调整原来头结点和第2个节点的指向
  newReverseListTail.next = head
  head.next = null

  // 将调整后的链表返回
  return newReverseList
}
```
