# [88. 合并两个有序数组](https://leetcode-cn.com/problems/merge-sorted-array/)

```js
var merge = function(A, m, B, n) {
  let i = m - 1, j = n - 1, k = m + n - 1;
  while(i >= 0 && j >= 0) {
    if (A[i] >= B[j]) {
      A[k] = A[i];
      i--;
      k--;
    } else {
      A[k] = B[j];
      j--;
      k--;
    }
  }
  while(j >= 0) {
    A[k] = B[j];
    k--;
    j--;
  }
};
```

var merge = function(nums1, m, nums2, n) {
  nums1.splice(m, n, ...nums2);
  nums1.sort((a, b) => a - b);
